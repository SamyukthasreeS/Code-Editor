C Conditional Statements

C provides if, else, and switch statements for decision making in your programs. 

// Example C conditional statements
int x = 10;
if (x > 5) {
    printf("x is greater than 5\n");
} else {
    printf("x is not greater than 5\n");
}

int choice = 2;
switch (choice) {
    case 1:
        printf("Option 1 selected\n");
        break;
    case 2:
        printf("Option 2 selected\n");
        break;
    default:
        printf("Invalid choice\n");
}

C Loops

C supports various types of loops like for, while, and do-while for repetitive tasks. 

// Example C loops
for (int i = 0; i < 5; i++) {
    printf("%d ", i);
}

int count = 0;
while (count < 3) {
    printf("%d ", count);
    count++;
}

int num = 0;
do {
    printf("%d ", num);
    num++;
} while (num < 3);

C Structures

C structures allow you to group different data types into a single variable. 

// Example C structures
struct Student {
    char name[50];
    int age;
    float GPA;
};

int main() {
    struct Student student1;
    strcpy(student1.name, "John");
    student1.age = 20;
    student1.GPA = 3.8;
    
    printf("Student Name: %s\n", student1.name);
    printf("Student Age: %d\n", student1.age);
    printf("Student GPA: %.2f\n", student1.GPA);
    return 0;
}

C Dynamic Memory

C allows you to allocate and deallocate memory dynamically using functions like malloc() and free(). 

// Example C dynamic memory allocation
#include 
#include 

int main() {
    int *arr = (int *)malloc(5 * sizeof(int));
    if (arr == NULL) {
        printf("Memory allocation failed\n");
        return 1;
    }
    for (int i = 0; i < 5; i++) {
        arr[i] = i * 2;
    }
    for (int i = 0; i < 5; i++) {
        printf("%d ", arr[i]);
    }
    free(arr);
    return 0;
}
